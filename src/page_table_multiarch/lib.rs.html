26/// The error type for page table operation failures.
27#[derive(Debug, PartialEq, Clone, Copy)]
28pub enum PagingError {
29    /// Cannot allocate memory.
30    NoMemory,
31    /// The address is not aligned to the page size.
32    NotAligned,
33    /// The mapping is not present.
34    NotMapped,
35    /// The mapping is already present.
36    AlreadyMapped,
37    /// The page table entry represents a huge page, but the target physical
38    /// frame is 4K in size.
39    MappedToHugePage,
40}
41
42#[cfg(feature = "axerrno")]
43impl From<PagingError> for axerrno::AxError {
44    fn from(value: PagingError) -> Self {
45        match value {
46            PagingError::NoMemory => axerrno::AxError::NoMemory,
47            _ => axerrno::AxError::InvalidInput,
48        }
49    }
50}

55/// The **architecture-dependent** metadata that must be provided for
56/// [`PageTable64`].
57pub trait PagingMetaData: Sync + Send {
58    /// The number of levels of the hardware page table.
59    const LEVELS: usize;
60    /// The maximum number of bits of physical address.
61    const PA_MAX_BITS: usize;
62    /// The maximum number of bits of virtual address.
63    const VA_MAX_BITS: usize;
64
65    /// The maximum physical address.
66    const PA_MAX_ADDR: usize = (1 << Self::PA_MAX_BITS) - 1;
67
68    /// The virtual address to be translated in this page table.
69    ///
70    /// This associated type allows more flexible use of page tables structs
71    /// like [`PageTable64`], for example, to implement EPTs.
72    type VirtAddr: MemoryAddr;
73    // (^)it can be converted from/to usize and it's trivially copyable
74
75    /// Whether a given physical address is valid.
76    #[inline]
77    fn paddr_is_valid(paddr: usize) -> bool {
78        paddr <= Self::PA_MAX_ADDR // default
79    }
80
81    /// Whether a given virtual address is valid.
82    #[inline]
83    fn vaddr_is_valid(vaddr: usize) -> bool {
84        // default: top bits sign extended
85        let top_mask = usize::MAX << (Self::VA_MAX_BITS - 1);
86        (vaddr & top_mask) == 0 || (vaddr & top_mask) == top_mask
87    }
88
89    /// Flushes the TLB.
90    ///
91    /// If `vaddr` is [`None`], flushes the entire TLB. Otherwise, flushes the
92    /// TLB entry at the given virtual address.
93    fn flush_tlb(vaddr: Option<Self::VirtAddr>);
94}

96/// The low-level **OS-dependent** helpers that must be provided for
97/// [`PageTable64`].
98pub trait PagingHandler: Sized {
99    /// Request to allocate a 4K-sized physical frame.
100    fn alloc_frame() -> Option<PhysAddr>;
101    /// Request to free a allocated physical frame.
102    fn dealloc_frame(paddr: PhysAddr);
103
104    /// Request to allocate contiguous physical frames with specified alignment.
105    ///
106    /// This is used for allocating page tables that require multiple pages or
107    /// specific alignment (e.g., ARMv7-A L1 page table needs 16KB with 16KB
108    /// alignment).
109    ///
110    /// # Arguments
111    ///
112    /// * `num_pages` - Number of 4K pages to allocate
113    /// * `align_pow2` - Alignment requirement in bytes (must be power of 2)
114    ///
115    /// Default implementation falls back to `alloc_frame()` for single 4K page.
116    fn alloc_frame_contiguous(num_pages: usize, align_pow2: usize) -> Option<PhysAddr> {
117        if num_pages == 1 && align_pow2 <= memory_addr::PAGE_SIZE_4K {
118            Self::alloc_frame()
119        } else {
120            None // Subclasses should override this for multi-page allocation
121        }
122    }
123
124    /// Request to free contiguous physical frames.
125    ///
126    /// # Arguments
127    ///
128    /// * `paddr` - Physical address of the first frame
129    /// * `num_pages` - Number of 4K pages to deallocate
130    ///
131    /// Default implementation falls back to `dealloc_frame()` for single page.
132    fn dealloc_frame_contiguous(paddr: PhysAddr, num_pages: usize) {
133        if num_pages == 1 {
134            Self::dealloc_frame(paddr);
135        }
136        // Subclasses should override this for multi-page deallocation
137    }
138
139    /// Returns a virtual address that maps to the given physical address.
140    ///
141    /// Used to access the physical memory directly in page table
142    /// implementation.
143    fn phys_to_virt(paddr: PhysAddr) -> VirtAddr;
144}

146/// The page sizes supported by the hardware page table.
147#[repr(usize)]
148#[derive(Debug, Copy, Clone, Eq, PartialEq)]
149pub enum PageSize {
150    /// Size of 4 kilobytes (2<sup>12</sup> bytes).
151    Size4K = 0x1000,
152    /// Size of 1 megabytes (2<sup>20</sup> bytes). Used by ARMv7-A.
153    Size1M = 0x10_0000,
154    /// Size of 2 megabytes (2<sup>21</sup> bytes).
155    Size2M = 0x20_0000,
156    /// Size of 1 gigabytes (2<sup>30</sup> bytes).
157    Size1G = 0x4000_0000,
158}
159
160impl PageSize {
161    /// Whether this page size is considered huge (larger than 4K).
162    pub const fn is_huge(self) -> bool {
163        matches!(self, Self::Size1G | Self::Size2M | Self::Size1M)
164    }
165
166    /// Checks whether a given address or size is aligned to the page size.
167    pub const fn is_aligned(self, addr_or_size: usize) -> bool {
168        memory_addr::is_aligned(addr_or_size, self as usize)
169    }
170
171    /// Returns the offset of the address within the page size.
172    pub const fn align_offset(self, addr: usize) -> usize {
173        memory_addr::align_offset(addr, self as usize)
174    }
175}
176
177impl From<PageSize> for usize {
178    #[inline]
179    fn from(size: PageSize) -> usize {
180        size as usize
181    }
182}

184/// This type indicates the mapping of a virtual address has been changed.
185///
186/// The caller can call [`TlbFlush::flush`] to flush TLB entries related to
187/// the given virtual address, or call [`TlbFlush::ignore`] if it knowns the
188/// TLB will be flushed later.
189#[must_use]
190pub struct TlbFlush<M: PagingMetaData>(M::VirtAddr, PhantomData<M>);
191
192impl<M: PagingMetaData> TlbFlush<M> {
193    pub(crate) const fn new(vaddr: M::VirtAddr) -> Self {
194        Self(vaddr, PhantomData)
195    }
196
197    /// Don't flush the TLB and silence the “must be used” warning.
198    pub fn ignore(self) {}
199
200    /// Flush the the TLB by the given virtual address to ensure the mapping
201    /// changes take effect.
202    pub fn flush(self) {
203        M::flush_tlb(Some(self.0))
204    }
205}
206
207/// This type indicates the page table mappings have been changed.
208///
209/// The caller can call [`TlbFlushAll::flush_all`] to flush the entire TLB, or
210/// call [`TlbFlushAll::ignore`] if it knowns the TLB will be flushed later.
211#[must_use]
212pub struct TlbFlushAll<M: PagingMetaData>(PhantomData<M>);
213
214impl<M: PagingMetaData> TlbFlushAll<M> {
215    pub(crate) const fn new() -> Self {
216        Self(PhantomData)
217    }
218
219    /// Don't flush the TLB and silence the “must be used” warning.
220    pub fn ignore(self) {}
221
222    /// Flush the entire TLB.
223    pub fn flush_all(self) {
224        M::flush_tlb(None)
225    }
